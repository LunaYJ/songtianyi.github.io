Lenny's Lucky Lotto Lists-DP

update@2011-08-03

for(i = 1;i <= 2000;i++){
    int j;
    dp[i][1] = i;
    for(j = 2;j <= 10 && ((1<<(j-1))) <= i;j++){
        int k;
        //dp[i][j] = dp[i>>1][j-1]
        dp[i][j] = 0;
        for(k = i;(k>>1) >= (1<<(j-2));k--){
            //dp[k>>1][j-1] + dp[(k-1)>>1][j-1]...... 这样相当于枚举了
            dp[i][j] += dp[k>>1][j-1];
        }
    }
    for(;j <= 10;j++)
        dp[i][j] = 0;
}
