A bug.s life-并查集

update@2011-09-22
C++语言: 高亮代码由发芽网提供

#define MAX 100010
int father[MAX];
int relation[MAX];
int makeSet(int N){
    //0表示和根节点的关系是 同类
    memset(relation,0,sizeof(relation));
    for(int i = 1;i <= N;i++){
        father[i] = i;
    }
    return 1;
}
int findSet(int x){
    if(x != father[x]){
        int temp = father[x];
        father[x] = findSet(father[x]);
        relation[x] = (relation[temp]+relation[x])%2;
    }
    return father[x];
}
int unionSet(int x,int y){
    int set_x = findSet(x);
    int set_y = findSet(y);
    if(set_x != set_y){
        father[set_x] = set_y;//将x并入y的集合
        relation[set_x] = ((relation[x]+1)%2+ relation[y])%2;
    }
    return 1;
}
